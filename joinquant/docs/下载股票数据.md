# 1. 进入项目目录
  cd /Users/yangqiang/Documents/aicode/quant/sharecode

  # 2. 激活环境
  conda activate aquant

  # 3. 执行（同时存 CSV + 写入数据库，限速 1.5 秒/次）
  python scripts/fetch_and_load_all.py --save-csv

  # 运行中断后可以用 --skip-existing 跳过已入库的标的继续
  python scripts/fetch_and_load_all.py --save-csv --skip-existing

  常用参数：

  ┌─────────────────────────┬──────────────────────────────┬────────┐
  │          参数           │             说明             │ 默认值 │
  ├─────────────────────────┼──────────────────────────────┼────────┤
  │ --delay 2.0             │ 请求间隔秒数，网络不稳时调大 │ 1.5    │
  ├─────────────────────────┼──────────────────────────────┼────────┤
  │ --retries 5             │ 单个标的最大重试次数         │ 5      │
  ├─────────────────────────┼──────────────────────────────┼────────┤
  │ --skip-existing         │ 已有数据的标的跳过不重写     │ 关     │
  ├─────────────────────────┼──────────────────────────────┼────────┤
  │ --save-csv              │ 同时把数据存到 data/ 目录    │ 关     │
  ├─────────────────────────┼──────────────────────────────┼────────┤
  │ --dry-run               │ 只拉数据不写库，用来测试     │ 关     │
  ├─────────────────────────┼──────────────────────────────┼────────┤
  │ --symbols 510300,512800 │ 只处理指定标的               │ 全部   │
  └─────────────────────────┴──────────────────────────────┴────────┘
